\paragraph{Task~A.}
Let binary variable~$Open_{j}$ take value~$1$ if a service station is built at location~$j$, and value~$0$ otherwise, with $j \in 1 \twodots z$ (set of zones).
Let binary variable~$Assign_{i,j}$ take value~$1$ if zone~$i$ is served by a vehicle from station~$j$, and value~$0$ otherwise, with $i,j \in 1 \twodots z$.

The objective is to minimize the total weighted time of service:
\[
  \text{minimize } \sum_{i \in Zones} \sum_{j \in Zones} (time_{i,j} \cdot weight_i) \cdot Assign_{i,j}
\]

Subject to the following constraints:
\begin{enumerate}
    \item \textbf{Demand Satisfaction:} Each zone~$i$ must be served by exactly~$c$ vehicles.
    \[
        \forall i \in Zones : \sum_{j \in Zones} Assign_{i,j} = c
    \]
    \item \textbf{Station Count:} Exactly~$s$ stations must be opened.
    \[
        \sum_{j \in Zones} Open_{j} = s
    \]
    \item \textbf{Capacity:} A station at zone~$j$ can provide at most~$v$ vehicles in total (per demand link).
    \[
        \forall i,j \in Zones : Assign_{i,j} \leq v \cdot Open_{j}
    \]
\end{enumerate}
